            app.ExitLumenRadiuscmEditField.Value = 0;

        % Value changed function: Stenosis1CheckBox
        function Stenosis1CheckBoxValueChanged(app, event)
            valuesPresentInTheEditFields = false;
            if (app.EntrySliceNumberEditField.Value > 0 || ...
                    app.ExitSliceNumberEditField.Value > 0 || ...
                    app.LumenNarrowingEditField.Value > 0 || ...
                    app.EntryLumenRadiuscmEditField.Value > 0 || ...
                    app.ExitLumenRadiuscmEditField.Value > 0)
                valuesPresentInTheEditFields = true;
            end
            value = app.Stenosis1CheckBox.Value;
            if (value == true)
                app.ExitLumenRadiuscmEditField.Enable = true;
                app.EntryLumenRadiuscmEditField.Enable = true;
                app.LumenNarrowingEditField.Enable = true;
                app.ExitSliceNumberEditField.Enable = true;
                app.EntrySliceNumberEditField.Enable = true;
                app.ExitLumenRadiuscmLabel.Enable = true;
                app.EntryLumenRadiuscmEditFieldLabel.Enable = true;
                app.LumenNarrowingEditFieldLabel.Enable = true;
                app.ExitSliceNumberEditFieldLabel.Enable = true;
                app.EntrySliceNumberEditFieldLabel.Enable = true;
                app.ClearButton.Enable = true;
                app.NumberofbranchbetweenprelesionandstenosisEditFieldLabel.Enable = true;
                app.NumberofbranchbetweenprelesionandstenosisEditField.Enable = true;
                app.EditField.Enable = true;
                app.EntryLabel.Enable = true;
                app.ExitLabel.Enable = true;
                app.numberOfStenosis = app.numberOfStenosis + 1;
                if (valuesPresentInTheEditFields)
                    deletePlotIDFromRemovalList(app, 1);
                end
            else
                app.ExitLumenRadiuscmEditField.Enable = false;
                app.EntryLumenRadiuscmEditField.Enable = false;
                app.LumenNarrowingEditField.Enable = false;
                app.ExitSliceNumberEditField.Enable = false;
                app.EntrySliceNumberEditField.Enable = false;
                app.ExitLumenRadiuscmLabel.Enable = false;
                app.EntryLumenRadiuscmEditFieldLabel.Enable = false;
                app.LumenNarrowingEditFieldLabel.Enable = false;
                app.ExitSliceNumberEditFieldLabel.Enable = false;
                app.EntrySliceNumberEditFieldLabel.Enable = false;
                app.ClearButton.Enable = false;
                app.NumberofbranchbetweenprelesionandstenosisEditFieldLabel.Enable = false;
                app.NumberofbranchbetweenprelesionandstenosisEditField.Enable = false;
                app.EditField.Enable = false;
                app.EntryLabel.Enable = false;
                app.ExitLabel.Enable = false;
                app.numberOfStenosis = app.numberOfStenosis - 1;
                if (valuesPresentInTheEditFields)
                    app.plotToRemove(end + 1) = 1;
                    app.plotToRemove= unique(app.plotToRemove);
                end
            end
        end
